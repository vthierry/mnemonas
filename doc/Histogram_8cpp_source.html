267 const Density& Histogram::getDensityModel(String model_) const
268 {
269  if(model_ == "gaussian") {
270  class GaussianDensity: public Density {
271  double m, s, k;
272 public:
273  GaussianDensity(double mean, double stdev) : m(mean), s(stdev), k(s * sqrt(2.0 * M_PI)) {}
274  double p(double x) const
275  {
276  if(0 < s) {
277  double u = (x - m) / s;
278  return exp(-0.5 * u * u) / k;
279  } else
280  return 1;
281  }
282  };
283  delete model;
284  model = new GaussianDensity(get ("mean"), get ("stdev"));
285  } else if(model_ == "gamma") {
286  class GammaDensity: public Density {
287  double d, t, k;
288 public:
289  GammaDensity(double degree, double rate) : d(degree), t(rate), k(t * tgamma(d)) {}
290  double p(double x) const
291  {
292  return 0 < t ? pow(x / t, d - 1) * exp(-x / t) / k : 1;
293  }
294  };
295  delete model;
296  model = new GammaDensity(get ("gamma-degree"), get ("gamma-rate"));
297  } else if(model_ == "uniform") {
298  double m = get("mean"), s = get("stdev");
299  class UniformDensity: public Density {
300  double min, max;
301 public:
302  UniformDensity(double min, double max) : min(min), max(max) {
303  assume(min <= max, "numerical-error", "in Histogram::get::UniformDensity incoherent uniform bounds [%g %g]", min, max);
304  }
305  double p(double x) const
306  {
307  return min < max ? 1 / (max - min) : 1;
308  }
309  };
310  delete model;
311  model = new UniformDensity(m - sqrt (3) *s, m + sqrt (3) *s);
312  } else if(model_ == "automatic")
313  switch((unsigned int) get("best-model")) {
314  case 1:
315  return getDensityModel("gaussian");
316  case 2:
317  return getDensityModel("gamma");
318  default:
319  return getDensityModel("uniform");
320  }
