142  q = r;
143  p = q;

146  q = (x0 - v) * (fx - fw);
147  p = (x0 - v) * q - (x0 - w) * r;

162  // F must not be evaluated too close to A or B.

169  } else {

176  }
177  // F must not be evaluated too close to X.
178  if(tol <= fabs(d))
179  u = x0 + d;
